from pandas import *

triangle = [[0], [0,1], [0,2]], [[0,1], [1], [1,2]], [[0,2], [1,2], [2]]
sub = [[0,0], [.8, .5],[1,0]]


def generate_base():
    df = concat({
        -1: concat({0: DataFrame(sub, range(3))})
    }, names=['level', 'triangle', 'node'])

    for level in range(2):
        for index in df.loc[level-1].index.get_level_values('triangle').unique():
            df = concat([df, concat({
                level: concat({
                    j + index*3: concat({
                        i: df.loc[level-1].loc[index].loc[pairs].mean()
                        for i, pairs in enumerate(t)
                    }) for j, t in enumerate(triangle)}).unstack(-1)})])
    return df

def generate(levels=3):
    df=generate_base()
    for q in range(levels):
        df = concat([
            df.add(_) for _ in sub
        ]).div([2, 2])
    return print("There are {} triangles.".format(df.pipe(len)/3)) or df
